# 最近更新

> 基于 GitHub 提交历史，自动拉取最新 30 条。

<div id="commits">加载中...</div>

<script>
(async () => {
  const el = document.getElementById('commits');
  try {
    const res = await fetch('https://api.github.com/repos/Nervger-lang/notes-blog/commits?per_page=30');
    const commits = await res.json();
    if (!Array.isArray(commits)) throw new Error('API limit');
    el.innerHTML = commits.map(c => {
      const d = new Date(c.commit.author.date);
      const date = `${d.getMonth()+1}/${d.getDate()} ${d.getHours().toString().padStart(2,'0')}:${d.getMinutes().toString().padStart(2,'0')}`;
      const msg = c.commit.message.split('\n')[0];
      const sha = c.sha.substring(0, 7);
      return `<div style="margin-bottom:8px;border-left:2px solid #5c6bc0;padding-left:10px">
        <span style="color:#999;font-size:0.85em">${date}</span>
        <a href="${c.html_url}" target="_blank" style="font-size:0.75em;color:#999;margin-left:6px">${sha}</a>
        <br>${msg}
      </div>`;
    }).join('');
  } catch(e) {
    el.innerHTML = '⚠️ 加载失败：' + e.message + '<br><a href="https://github.com/Nervger-lang/notes-blog/commits/main" target="_blank">→ 在 GitHub 查看</a>';
  }
})();
</script>
